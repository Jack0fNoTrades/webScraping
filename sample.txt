    function ScrollContactFormUpdate() {

            }
        });

            contactFormObj.addClass('affix');

            var contactWindowHeight = contactFormObj.outerHeight() + reportLinkSpanHeight;
            var adHeight = (adObj.length > 0 ? adObj.outerHeight() : 0) + 20;

            if (adObj.length > 0) {
                if (contactWindowHeight + adHeight + (brokerObj.is(':visible') ? 0 : brokerHeight) >= windowHeight - 20) {
                    adObj.removeClass('affix');
                }
                else {
                    if (!noPhotoAvailable)
                    {
                        adObj.addClass('affix');
                    }
                }
            }

            if (contactWindowHeight + (brokerObj.is(':visible') ? 0 : (brokerHeight + 35)) >= windowHeight - 20) {
                if (adObj.length > 0) {
                    adObj.removeClass('affix');
                }

                contactWindowHeight = contactFormObj.outerHeight();

            bottomPos = bottomPosObj.offset().top - contactWindowHeight - 35 - adHeight;

                topOfControl = bottomPos - window.pageYOffset;
            }

            adObj.css({ 'top': (topOfControl + contactWindowHeight) + 'px' });
        }
        else {
            contactFormObj.removeClass('affix');

            if (adObj.length > 0) {
                adObj.removeClass('affix');
            }
        }
    }
